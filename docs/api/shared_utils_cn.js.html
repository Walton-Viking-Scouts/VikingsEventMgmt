/**
 * Utility function for conditionally joining class names

 * 
 * Examples:
 * cn('base', condition && 'conditional', { 'active': isActive })
 * cn(['base', 'other'], { 'variant': true })
 */

export function cn(...classes) {
  return clsx(classes);
}
